// 月齢と月齢アイコンは全天体で辻時刻の月の状態を表示
const phase = Astronomy.MoonPhase(dt);

const icons = ['🌑', '🌒', '🌓', '🌔', '🌕', '🌖', '🌗', '🌘'];
